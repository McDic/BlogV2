    Template¶

    Generic programmingмқҙлһҖ, м–ҙл–Ө н–үлҸҷмқ„ н•ҳлҠ” мҪ”л“ңк°Җ мһ‘м„ұлҗҳм—Ҳмқ„ л•Ң к·ё мҪ”л“ңм—җм„ң н•„мҡ”н•ң ліҖмҲҳ, н•ЁмҲҳ л“ұмқҳ н–үлҸҷліҙлӢӨ нғҖмһ…мқ„ лӮҳмӨ‘м—җ кІ°м •н•ҳлҠ” н”„лЎңк·ёлһҳл°Қ нҢЁлҹ¬лӢӨмһ„мқ„ л§җн•©лӢҲлӢӨ. C++м—җм„  мқҙкІғмқҙ templateлҘј нҶөн•ҙ к°ҖлҠҘн•ңлҚ°мҡ”. мҳҲлҘј л“Өм–ҙ лӢӨмқҢкіј к°ҷмҠөлӢҲлӢӨ.

    м—¬кё°м„ң convert_to_int н•ЁмҲҳлҠ” ліөмҲҳмқҳ нғҖмһ… Tм—җ лҢҖн•ҙм„ң T нғҖмһ…мқ„ к°Җм§ҖлҠ” к°қмІҙлҘј н•ҳлӮҳ л°ӣм•„ intлҘј лҰ¬н„ҙн•ҙмЈјлҠ” н•ЁмҲҳк°Җ лҗ©лӢҲлӢӨ.

    C++м—җм„ң templateлҘј нҶөн•ҙм„ң genericн•ҳкІҢ л§Ңл“Ө мҲҳ мһҲлҠ” мҡ”мҶҢл“ӨмқҖ лӢӨмқҢкіј к°ҷмҠөлӢҲлӢӨ.

    • н•ЁмҲҳ (function template)
    • нҒҙлһҳмҠӨ (class template)
    • alias (alias template)
    • ліҖмҲҳ (variable template)

    к·ё мҷё concept templateлҸ„ мһҲм§Җл§Ң, мқҙкұ°лҠ” C++20м—җм„ң лӢӨлЈЁлҠ” лӮҙмҡ©мқҙкё° л•Ңл¬ём—җ мғқлһөн•ҳлҸ„лЎқ н•ҳкІ мҠөлӢҲлӢӨ.

    Function Templates¶

    templateлҘј мӮ¬мҡ©н•ҳл©ҙ Genericн•ң нғҖмһ…мқҳ нҢҢлқјлҜён„° лҳҗлҠ” лҰ¬н„ҙ нғҖмһ…мқ„ к°Җм§ҖлҠ” н•ЁмҲҳлҘј л§Ңл“Ө мҲҳ мһҲмҠөлӢҲлӢӨ. Template н•ЁмҲҳлҘј м •мқҳн•ҳлҠ” syntaxлҠ” лӢӨмқҢкіј к°ҷмҠөлӢҲлӢӨ.

    template <...> return_type function_name (...);

    лӢӨмқҢмқҖ template function м •мқҳмқҳ мҳҲмӢңмһ…лӢҲлӢӨ.

    #include <iostream>
#include <string>

template <class T> void swap(T &obj1, T &obj2) {
    T temp = obj1;
    obj1 = obj2;
    obj2 = temp;
}

int main(void) {
    int a = 1, b = 2;
    double c = 3.4, d = 5.6;
    std::string e = "ee", f = "ff";
    swap(a, b); // swap<int>
    swap(c, d); // swap<double>
    swap(e, f); // swap<std::string>
    std::cout << a << b << c << d << e << f << std::endl;
}

    Template can't do everything¶

    templateмқҙ genericн•ң н•ЁмҲҳлҘј л§Ңл“Өм–ҙмӨ¬лӢӨкі  н•ҙм„ң мӢӨм ңлЎң л¶Ҳк°ҖлҠҘн•ң м—°мӮ°к№Ңм§Җ мҳ¬л°”лҘҙкІҢ л§Ңл“Өм–ҙмЈјлҠ” кІғмқҖ м•„лӢҷлӢҲлӢӨ. мҳҲлҘј л“Өм–ҙліҙкІ мҠөлӢҲлӢӨ.

    return obj + 1;
}

int main(void) {
    // Number arithmetic
    add_one(1);
    add_one(-2.34);

    // It's ok because of pointer arithmetic..
    add_one("const char*");

    // Compile Error!
    // add_one(std::string("dd"));
}

    мң„ мҪ”л“ңм—җм„ң int, double, к·ёлҰ¬кі  const char* нғҖмһ…мқҖ intмҷҖмқҳ лҚ§м…Ҳ м—°мӮ°мқҙ м§Җмӣҗлҗҳкё° л•Ңл¬ём—җ мғҒкҙҖм—Ҷм§Җл§Ң, std::stringмқҖ + 1мқҙ м•Ҳ лҗҳкё° л•Ңл¬ём—җ м»ҙнҢҢмқјлҹ¬к°Җ instantiationмқ„ л§Ңл“ңлҠ” кіјм •м—җм„ң м—җлҹ¬к°Җ л°ңмғқн•©лӢҲлӢӨ. (instantiationм—җ кҙҖн•ҙм„ңлҠ” нӣ„мҲ н•ҳкІ мҠөлӢҲлӢӨ.)

    Template argument deduction¶

    Function templateлҘј instantiateн•ҳкё° мң„н•ҙм„ңлҠ” лӘЁл“  template argumentлҘј м»ҙнҢҢмқјлҹ¬к°Җ м•Ңм•„м•ј н•ҳм§Җл§Ң, н”„лЎңк·ёлһҳлЁёк°Җ л°ҳл“ңмӢң мқҙлҘј лӘ…мӢңм ҒмңјлЎң м§Җм •н•ҙмӨ„ н•„мҡ”лҠ” м—ҶмҠөлӢҲлӢӨ. лӢӨмқҢ мҪ”л“ңмІҳлҹј мқјл¶Җ н…ңн”ҢлҰҝ нҢҢлқјлҜён„°л§Ң м§Җм •н•ҙмЈјлҚ”лқјлҸ„(лҳҗлҠ” м•„мҳҲ м§Җм •н•ҳм§Җ м•ҠлҚ”лқјлҸ„), м»ҙнҢҢмқјлҹ¬лҠ” м§Җм •лҗҳм§Җ м•ҠмқҖ н…ңн”ҢлҰҝ нҢҢлқјлҜён„°лҘј м•Ңм•„м„ң м¶”м •н•ҙлӮј мҲҳ мһҲлӢӨл©ҙ к·ё м¶”м •н•ң нғҖмһ… лҳҗлҠ” к°’мқ„ н…ңн”ҢлҰҝ нҢҢлқјлҜён„°лЎң мӮ¬мҡ©н•©лӢҲлӢӨ.

    #include <string>
#include <utility>

template <class T1, class T2, class T3>
std::pair<T1, T2> pairize(T3 x) {
    return {(T1)x, (T2)x};
}

int main(void) {
    // pairize<float, double, int>
    std::pair<float, double> a = pairize<float, double>(-1);
    // pairize<std::string, std::string, const char*>
    auto [b1, b2] = pairize<std::string, std::string>("abc");

    std::pair<int, long double> (*c) (float) = pairize;

    Template argument substitution¶

    л§Ңм•Ҫ лӘЁл“  template argumentк°Җ лӘ…мӢңм ҒмңјлЎң м§Җм •лҗҳм—ҲлӢӨл©ҙ, к·ё м•Ҳм—җ л“Өм–ҙк°ҖлҠ” нҢҢлқјлҜён„°л“ӨмқҖ н•ЁмҲҳ/нҒҙлһҳмҠӨ лӮҙл¶Җм—җм„ң н•ҙлӢ№ нғҖмһ…мңјлЎң кі м •мқҙ лҗ©лӢҲлӢӨ. м ҖлҠ” нҢҢлқјлҜён„°л“Өм—җ лҢҖн•ҙм„ң implicit type castingмқҙ мқјм–ҙлӮңлӢӨкі  мқҙн•ҙн–ҲмҠөлӢҲлӢӨ.

    кІҪмҡ°м—җ л”°лқј substitutionмқҙ мӢӨнҢЁн•ҳлҠ” кІҪмҡ°к°Җ мһҲлҠ”лҚ°, к·ёкІғмқҙ м»ҙнҢҢмқј м—җлҹ¬лЎң мқҙм–ҙм§Җм§ҖлҠ” м•ҠмҠөлӢҲлӢӨ. мқҙлҘј SFINAE (Substitution Failure Is Not An Error) лқјкі  н•©лӢҲлӢӨ. л§Ңм•Ҫ лӘ…мӢңм ҒмңјлЎң м§Җм •лҗҳм—Ҳкұ°лӮҳ лҳҗлҠ” deducedлҗң н…ңн”ҢлҰҝ нҢҢлқјлҜён„°лҘј substitution н•ҳлҠ” кІҢ мӢӨнҢЁн•ҳл©ҙ, н•ҙлӢ№ specializationмқҙ лІ„л Өм§ҖлҠ” к·ңм№ҷмқ„ мқҳлҜён•©лӢҲлӢӨ.

    лӢӨмқҢ мҪ”л“ңмҷҖ к°ҷмқҙ, top-level CV qualifierмқҳ м ңкұ°лҠ” н•ҙлӢ№ н•ЁмҲҳмқҳ нҢҢлқјлҜён„° нғҖмһ…м—җ мҳҒн–Ҙмқ„ лҜём№ҳм§Җ м•ҠмҠөлӢҲлӢӨ. (лӢӨл§Ң, GCC Bug 85428м—җ мқҳн•ҙ gcc 12 лҜёл§Ңм—җм„ң м»ҙнҢҢмқј лҗҳм§Җ м•ҠмҠөлӢҲлӢӨ!)

    template <class T> void f(T t) {}
template <class T> void g(const T t) {}
template <class T> void h(T t, T* tp) {}

int main(void) {
    static_assert(main == main); // OK

    constexpr auto fint = f<int>; // void(int), t: int
    constexpr auto fconstint = f<const int>; // void(int), t: const int
    static_assert(fint != fconstint); // FAILS DUE TO BUG

    constexpr auto gint = g<int>; // void(int), t: const int
    constexpr auto gconstint = g<const int>; // void(int), t: const int
    static_assert(gint != gconstint); // FAILS DUE TO BUG

    constexpr auto hint = h<int>; // void(int, int*)
    constexpr auto hconstint = h<const int>; // void(int, const int*)
}

    Function template overloading¶

    Function templateмқҖ мҳӨлІ„лЎңл”© лҗ  мҲҳ мһҲмҠөлӢҲлӢӨ.

    мқјл°ҳ н•ЁмҲҳлҠ” н•ӯмғҒ(лҰ¬н„ҙ нғҖмһ…кіј нҢҢлқјлҜён„° нғҖмһ…мқҙ лӘЁл‘җ к°ҷлҚ”лқјлҸ„!) template functionкіј кө¬л¶„лҗ©лӢҲлӢӨ. (always distinct) лҸҷмқјн•ң мқҙлҰ„мқҳ мқјл°ҳ н•ЁмҲҳк°Җ м •мқҳлҗҳм–ҙ мһҲлҠ” кІҪмҡ°, лҸҷмқјн•ң нғҖмһ…мқҳ н…ңн”ҢлҰҝ н•ЁмҲҳлҘј нҳём¶ңн•ҳкі  мӢ¶лӢӨл©ҙ explicitн•ҳкІҢ template parameterлҘј м •мқҳн•ҙмӨҳм•ј н•©лӢҲлӢӨ.

    #include <iostream>

template <class T> T f(T x) { return x; }
int f(int x) { return x+1; }

int main(void) {
    // Prints "1 2"
    std::cout << f<int>(1) << ' ' << f(1) << std::endl;
}

    лӢӨмқҢ мҪ”л“ңмҷҖ к°ҷмқҙ нҢҢлқјлҜён„° нғҖмһ…мқ„ лӘЁл‘җ лӘ…мӢңм ҒмңјлЎң м§Җм •н•ҳлҚ”лқјлҸ„ return нғҖмһ…мқҙ ambiguousн•ҙм„ң м»ҙнҢҢмқјлҹ¬к°Җ deduceн•ҳм§Җ лӘ»н•ҳлҠ” кІҪмҡ°к°Җ мһҲмҠөлӢҲлӢӨ. мқҙлҹҙ кІҪмҡ° лҰ¬н„ҙ нғҖмһ…к№Ңм§Җ лӘЁл‘җ лӘ…мӢңн•ҙмӨҳм•ј н•©лӢҲлӢӨ. н•ҳм§Җл§Ң, нҢҢлқјлҜён„° нғҖмһ…лҸ„ к°ҷкі  лҰ¬н„ҙ нғҖмһ…к№Ңм§Җ к°ҷмқҖ кІҪмҡ°, ODL Ruleм—җ мқҳн•ҙм„ң м •мқҳлҘј 2лІҲ мқҙмғҒ н•  мҲҳ м—ҶмҠөлӢҲлӢӨ.

    template <int I> struct A {
    const static int member = I;
};

template <int I, int J> A<I+J> f(A<I>, A<J>) { return A<I+J>(); }
// template <int K, int L> A<K+L> f(A<K>, A<L>) { return A<K+L>(); } // Compile Error!
template <int I, int J> A<I-J> f(A<I>, A<J>) { return A<I-J>(); }

int main(void) {
    // auto f23 = f<2, 3>; // Compile Error!
    A<5> (*f2p3)(A<2>, A<3>) = f<2, 3>;
    A<5> (*f1p4)(A<1>, A<4>) = f<1, 4>;
    A<-1> (*f2m3)(A<2>, A<3>) = f<2, 3>;
}

    л‘җ function templatesлҠ” лӢӨмқҢ мЎ°кұҙмқ„ лӘЁл‘җ л§ҢмЎұн•  л•Ң equivalentн•ҳлӢӨкі  н‘ңнҳ„н•©лӢҲлӢӨ.

    • лҸҷмқјн•ң scope лӮҙм—җм„ң declared лҗҳм–ҙ мһҲмқ„ л•Ң
    • лҸҷмқјн•ң н•ЁмҲҳ мқҙлҰ„мқ„ к°Җм§Ҳ л•Ң
    • лҸҷмқјн•ң template parameter listлҘј к°Җм§Ҳ л•Ң(ex: <int, double, A<int>>)
    • template parameter кҙҖл Ён•ҙм„ң мӮ¬мҡ©лҗҳлҠ” expressionл“Өмқҳ лҰ¬н„ҙ нғҖмһ…кіј нҢҢлқјлҜён„° лҰ¬мҠӨнҠёк°Җ лӘЁл‘җ лҸҷмқјн•  л•Ң

    Template parameterм—җ мӮ¬мҡ©лҗҳлҠ” л‘җ expressionмқҖ лӢӨмқҢ мЎ°кұҙмқ„ лӘЁл‘җ л§ҢмЎұн•  л•Ң functionally equivalentн•ҳлӢӨкі  н‘ңнҳ„н•©лӢҲлӢӨ.

    • л‘җ expressionмқҙ equivalentн•ҳм§Җ м•Ҡмқ„ л•Ң
    • лҸҷмқјн•ң н…ңн”ҢлҰҝ нҢҢлқјлҜён„°м—җ лҢҖн•ҙм„ң л‘җ expressionмқ„ мӢӨн–үн•ң к°’мқҙ к°ҷмқ„ л•Ң

    л§Ңм•Ҫ лӢӨмқҢ мҪ”л“ңмҷҖ к°ҷмқҙ н”„лЎңк·ёлһЁмқҙ functionally equivalentн•ҳм§Җл§Ң equivalentн•ҳм§Җ м•ҠмқҖ function templateлҘј нҸ¬н•Ён•ҳкі  мһҲлӢӨл©ҙ, н•ҙлӢ№ н”„лЎңк·ёлһЁмқҖ ill-formed лҗҳм—ҲлӢӨкі  к°„мЈјлҗ©лӢҲлӢӨ.

template <int I> void f(A<I>, A<I+10>); // #1

template <int I> void f(A<I>, A<I+1+2+3+4>) {} // #2; Functionally equivalent to #1

int main(void) {
    // f(A<1>{}, A<11>{}); // Compile Error!
}

    Partial Ordering¶

    мЈјмқҳ: м•„м§Ғ н•ҙлӢ№ мӮ¬н•ӯмқҙ function specializationм—җ лҢҖн•ҙ мһ‘лҸҷн•ҳлҠ” кұҙм§Җ, function overloadingм—җ лҢҖн•ҙ мһ‘лҸҷн•ҳлҠ” кұҙм§Җ, м•„лӢҲл©ҙ л‘ҳ лӢӨмқём§Җ м•„м§Ғ мқҙн•ҙлҘј лӘ»н–ҲмҠөлӢҲлӢӨ. л¬ём„ңм—җлҠ” specializationмӘҪм—җ лҢҖн•ҙм„ң мҚЁм ё мһҲм–ҙм„ң specializationм—җ лҢҖн•ҙ мһ‘лҸҷн•ңлӢӨкі  лҗҳм–ҙ мһҲлҠ” кІғ к°ҷкёҙ н•ңлҚ°, к·ёлҹј л‘ҳмқҙ exclusive mutualн•ң к°ңл…җмқҙ м•„лӢҢ кұҙк°Җ? лқјлҠ” мғқк°ҒлҸ„ л“Өкі .. м•„л¬ҙнҠј мўҖ лҚ” м •лҰ¬лҗҳл©ҙ м •нҷ•н•ҳкІҢ м •лҰ¬н•ҙліҙкІ мҠөлӢҲлӢӨ.

    C++м—җм„ң м—¬лҹ¬ function template specializationм—җ лҢҖн•ҙм„ң м–ҙл–Ө н•ЁмҲҳлҘј мӮ¬мҡ©н• м§Җ м„ нғқн•ҳлҠ” кіјм •мқ„ Partial Orderingмқҙлқјкі  н•©лӢҲлӢӨ.

    1. л‘җ function templateмқ„ к°Ғк°Ғ T1, T2лқјкі  н•©мӢңлӢӨ.
    2. T1м—җ м Ғмҡ©н•  мҲҳ мһҲлҠ” нҢҢлқјлҜён„° нғҖмһ…мқ„ T2м—җ лҢҖмһ…н•ҙліҙкі , л°ҳлҢҖлЎңлҸ„ н•ҙлҙ…лӢҲлӢӨ.
    3. 2лІҲмқҳ кіјм •мқҙ н•ңмӘҪл§Ң м„ұкіөн•ңлӢӨл©ҙ, more specialized мҡ°м—ҙмқҙ кІ°м •лҗ©лӢҲлӢӨ. (л§Ңм•Ҫ м–‘мӘҪмқҙ м„ұкіөн•ңлӢӨл©ҙ, к·ё н•ЁмҲҳлҘј мҪңн•ҳл Өкі  н•  л•Ң ambiguous callмқҙ лҗҳм–ҙлІ„лҰҪлӢҲлӢӨ.)

    лӢӨмқҢ мҪ”л“ңлҠ” C++ Referenceм—җ мқҳн•ҳл©ҙ м»ҙнҢҢмқјмқҙ лҗҳм–ҙм•ј н•  кІғ к°ҷмқҖлҚ°, м•Ҳ лҗҳл„Өмҡ”.. (gcc 7, 10, 11 лӢӨ м•Ҳ лҗ©лӢҲлӢӨ)

    #include <iostream>

struct A {};

template <class T> struct B {
    template <class R> int operator* (R&) { return 0; }
}; // More specialized

template <class T, class R> int operator* (T&, R&) { return 1; }

int main() {
    A a;
    B<A> ba;
    std::cout << ba * a << std::endl;
}

    л‘җ template specialization A, Bм—җ лҢҖн•ҙ мқјл¶Җ мЎ°кұҙмқ„ л§ҢмЎұн•ҳл©ҙ A is more specialized than B лқјкі  н•ҳлҠ”лҚ°, к·ёлҹ° мЎ°кұҙл“Өмқҳ лӘ©лЎқмқҖ лӢӨмқҢкіј к°ҷмҠөлӢҲлӢӨ.

    • T*мқҖ TліҙлӢӨ лҚ” specialized н•ҳкі , const TлҠ” TліҙлӢӨ лҚ” specialized н•ҳкі , const T*лҠ” T*ліҙлӢӨ лҚ” specialized н•©лӢҲлӢӨ.

template <class T> const char* f(T) { return "f(T)"; }

    std::cout << f(x) << std::endl;
}
    • Reference нғҖмһ…мқҳ кІҪмҡ° pointerмҷҖлҠ” лӢӨлҘҙкІҢ лҸҷмһ‘н•©лӢҲлӢӨ. TмҷҖ T&, TмҷҖ T&&лҠ” ambiguousн•©лӢҲлӢӨ.

template <class T> const char* f(T&&) { return "f(T&&)"; }

int main(void) {
    // std::cout << f(1.0) << std::endl; // Ambiguous between T and T&&
    double x;
    // std::cout << f(x) << std::endl; // Ambiguous between T and T&
}
    • лӘ…мӢңм ҒмңјлЎң м§Җм •лҗң нғҖмһ…кіј н…ңн”ҢлҰҝмңјлЎң deduceлҗң нғҖмһ…мқҙ лҸҷмқјн•ҙм„ң л‘җ н…ңн”ҢлҰҝмқҙ кІ№м№ҳлҠ” кІҪмҡ°к°Җ мһҲмҠөлӢҲлӢӨ. мқҙ кІҪмҡ°лҠ” ambiguousн•©лӢҲлӢӨ. 
      #include <iostream>

template <class T> const char* f(T, T) { return "f(T, T)"; }
template <class T> const char* f(T, int) { return "f(T, int)"; }

int main(void) {
    // const char* (*func)(int, int) = f<int>; // Compile Error!
}
    • н•ЁмҲҳ мҪң м»Ён…ҚмҠӨнҠё(call context)лҠ” мҳӨм§Ғ explicit call argumentsм—җ мқҳн•ҙм„ңл§Ң мўҢмҡ°лҗ©лӢҲлӢӨ. л”°лқјм„ң Default function parameter, function parameter packs, ellipsis parameter л“ұмқҖ лӘ…мӢңн•ҳм§Җ м•ҠлҠ” мқҙмғҒ л¬ҙмӢңлҗ©лӢҲлӢӨ. 
      template <class T> void f(T) {}
template <class T> void f(T, int i=2) {}
template <class T> void f(T, ...) {}

    f(1, 2, 3); // OK
    // f(1); // Compile Error!
}

    Function overload is not a function specialization¶

    Function overloadлҠ” non-templateмҷҖ primary-templateм—җ мқҳн•ҙм„ңл§Ң кІ°м •лҗҳл©°, к·ё кіјм •м—җм„ң function specializationмқҖ л¬ҙмӢңлҗ©лӢҲлӢӨ. лӢӨмқҢ мҪ”л“ңлҠ” к·ё мҳҲмӢңмһ…лӢҲлӢӨ.

    #include <iostream>

template <class T> void f(T) { // #1 Primary
    std::cout << "f<T>(T)\n";
}
template <> void f(int*) { // #1 specialization
    std::cout << "f<int>(int*)\n";
}
template <class T> void f(T*) { // #2 Primary (overload)
    std::cout << "f<T>(T*)\n";
}

int main(void) {
    int *x = new int;
    f(x); // Calls #2 Primary
    delete x;
}

    Class Templates¶

    Template classлҘј м •мқҳн•ҳлҠ” syntaxлҠ” лӢӨмқҢкіј к°ҷмҠөлӢҲлӢӨ.

    template <...> class_name;

    мҳҲлҘј л“Өл©ҙ лӢӨмқҢкіј к°ҷмҠөлӢҲлӢӨ.

    #include <iostream>

template <class T> class A {
private:
    T member;
public:
    A() = delete;
    A(T x): member(x) {}
    T get() { return member; }

    template <class X> struct Local {
        static X convert(T wow) {
            return X(wow);
        }
    };
};

int main(void) {
    std::cout << A(3).get() << '\n'
              << A(1.23).get() << '\n'
              << A("bbb").get() << '\n'
              << A<int>::Local<double>::convert(1.2) << '\n';
}

    мғҒмҶҚлҸ„ мһҗмң лЎӯкІҢ н•  мҲҳ мһҲмҠөлӢҲлӢӨ. кІҪмҡ°м—җ л”°лқј primary templateмқ„ к·ёлҢҖлЎң мғҒмҶҚл°ӣлҚҳк°Җ, нҠ№м • instantiationл§Ң м§Җм •н•ҙм„ң мғҒмҶҚл°ӣмқ„ мҲҳ мһҲмҠөлӢҲлӢӨ.

    template <class T> struct L1 { T member; };
template <class T> struct L2: L1<T> { T member2; };
template <class T> struct L2fromInt32: L1<int> {};

int main(void) {
    L1<double> {};
    L2<long long> {};
    L2fromInt32<const char*> {};
}

    Specializations can have different configurations¶

    лӢӨмқҢ мҪ”л“ңмҷҖ к°ҷмқҙ, template classм—җ лҢҖн•ҙ specializationмқ„ л§Ңл“Ө кІҪмҡ° н•ҙлӢ№ template classмқҳ primary templateм—җ мһҲлҠ” лӘЁл“  кІғмқ„ м „л¶Җ мһ¬м •мқҳн•ҙмӨҳм•ј н•©лӢҲлӢӨ. мқҙ л¶Җл¶„мқҖ мўҖ л¶ҲнҺён•ң кІғ к°ҷл„Өмҡ”.

    Class Template Argument Deduction (CTAD)¶

    Function template лҝҗл§Ң м•„лӢҲлқј, class templateм—җ лҢҖн•ҙм„ңлҸ„ template argument deductionмқҙ мһ‘мҡ©н•©лӢҲлӢӨ. мқҙлҘј CTADлқјкі  н•©лӢҲлӢӨ.

    Implicit CTAD¶

    н”„лЎңк·ёлһҳлЁёк°Җ нҠ№лі„нһҲ м§Җм •н•ҙмЈјм§Җ м•Ҡм•„лҸ„, мқҙлҜё м •мқҳлҗҳм–ҙ мһҲлҠ” мғқм„ұмһҗ лҳҗлҠ” мһҗлҸҷмңјлЎң мғқм„ұлҗҳлҠ” copy/move constructorлҘј нҶөн•ҙ deduceлҗҳлҠ” кІғмқ„ implicit CTADлқјкі  н•©лӢҲлӢӨ.

    template <class T> struct A {
    A(T, T) {}
};

int main(void) {
    A a = A{1.0, 2.0};
    auto a2 = A(a);
    auto a_ptr = new A{1, 2};
    delete a_ptr;
}

    User-defined CTAD¶

    лӢӨмқҢкіј к°ҷмқҖ syntaxлҘј мӮ¬мҡ©н•ҳм—¬ м§Ғм ‘ CTADлҘј м •мқҳн•  мҲҳ мһҲмҠөлӢҲлӢӨ.

    template_name (...) -> template_id;

    л¬јлЎ  мқҙ syntax м•һм—җ template <...> лҘј л¶ҷм—¬м„ң CTADлҘј genericн•ҳкІҢ м •мқҳн•ҳлҠ” кІғ лҳҗн•ң к°ҖлҠҘн•©лӢҲлӢӨ. лҳҗн•ң, explicitмқҙ template_name м•һм—җ optionalн•ҳкІҢ л¶ҷмқ„ мҲҳ мһҲлҠ”лҚ°мҡ”. мқҙ кІҪмҡ° кё°мЎҙ нҒҙлһҳмҠӨ constructorм—җ explicitмқҙ л¶ҷлҠ” нҡЁкіјмҷҖ лҸҷмқјн•ң нҡЁкіјк°Җ мһ‘мҡ©н•©лӢҲлӢӨ. (Copy initializationмқҙ н—Ҳмҡ©лҗҳм§Җ м•ҠмҠөлӢҲлӢӨ) лӢӨмқҢ мҪ”л“ңлҠ” CTADлҘј м§Ғм ‘ м •мқҳн•ҳлҠ” мҳҲмӢңмһ…лӢҲлӢӨ.

    #include <string>

template <class T1, class T2> struct Employee {
    T1 name; T2 salary;
    Employee(T1 &name_): name(name_), salary(0) {}
    Employee(T1 &&name_): name(name_), salary(0) {}
    Employee(T1 name_, T2 salary_): name(name_), salary(salary_) {}
};

explicit Employee(const char*, int) -> Employee<std::string, double>;
// template <class T> explicit Employee(T) -> Employee<T, double>;
template <class T> Employee(T&) -> Employee<T, double>;

int main(void) {

    // <std::string, double> instead of <const char*, int>
    auto e1 = Employee("McDic", 1234);
    // Employee e1b = {"McDic", 1234}; // Compile Error! (explicit)

    // <T, double> where T = const char[6]
    auto e2 = Employee("McDic");
    Employee e2b = "McDic"; // OK, non-explicit

    char name = 'M';
    auto e3 = Employee(name);
    // auto e4 = Employee('M'); // Compile Error! (Can't deduce T2)
}

    Alias Templates¶

    using, typedefлҘј нҷңмҡ©н•ҳм—¬ template typeм—җ лҢҖн•ң aliasлҘј нҷңмҡ©н•  мҲҳ мһҲмҠөлӢҲлӢӨ.

    template <class T> struct Lout {
    struct Lin1 { T member; };
    template <class U> struct Lin2 {
        T member2a;
        U member2b;
    };
};

template <class T> using Lin1Alias = typename Lout<T>::Lin1;
template <class T, class U> using Lin2Alias = typename Lout<T>::template Lin2<U>;

int main(void) {
    ::Lin1Alias<int> x;
    ::Lin2Alias<double, const char*> y;
}

    Template class лӮҙл¶Җмқҳ template classлҘј м ‘к·јн•ҳл Өл©ҙ м Җлҹ° мӢқмңјлЎң ::templateлҘј н•ҙмӨҳм•ј н•ҳлҠ”лҚ°, мҷң к·ёлҹ°м§ҖлҠ” м•„м§Ғ мһҳ лӘЁлҘҙкІ мҠөлӢҲлӢӨ..

    Variable Templates¶

    C++14л¶Җн„°лҠ” ліҖмҲҳм—җлҸ„ н…ңн”ҢлҰҝмқ„ м”Ңмҡ°лҠ” кІҢ к°ҖлҠҘн•ҙмЎҢмҠөлӢҲлӢӨ. к·ё м „м—җлҸ„ static data memberлӮҳ constexpr function template л“ұмңјлЎң мӮ¬мӢӨмғҒ н…ңн”ҢлҰҝ ліҖмҲҳк°Җ к°ҖлҠҘн•ҳкёҙ н–Ҳм§Җл§Ң, м–ҙмЁҢл“  н•ҳлӮҳмқҳ м§ҒкҙҖм Ғмқё л°©лІ•мқҙ к°ҖлҠҘн•ҙм§„ м…Ҳмһ…лӢҲлӢӨ.

    syntaxлҠ” лӢӨмқҢкіј к°ҷмҠөлӢҲлӢӨ.

    template <...> variable_name;

    мҳҲмӢңлҘј л“Өм–ҙліҙкІ мҠөлӢҲлӢӨ.

    #include <iostream>

template <class T> T pi = T(3.14);

    Template Specialization¶

    Template parameterлЎң нҠ№м • к°’ лҳҗлҠ” нғҖмһ…мқ„ л„Јм—Ҳмқ„ л•Ңл§Ң м–ҙл–Ө н–үлҸҷмқ„ лӢӨлҘҙкІҢ н•ҳкі  мӢ¶мқ„ мҲҳ мһҲмҠөлӢҲлӢӨ. к·ёкІҢ к°ҖлҠҘн•ңлҚ°, к·ёкұё template specializationмқҙлқјкі  н•©лӢҲлӢӨ. Template functionм—җм„ңлҸ„ мһ к№җ м„ӨлӘ…л“ңл ём§Җл§Ң, мқҙкұҙ overloadingкіјлҠ” мЎ°кёҲ лӢӨлҘё к°ңл…җмһ…лӢҲлӢӨ.

    Specializationм—җлҠ” л‘җ мў…лҘҳк°Җ мһҲлҠ”лҚ°, н•ҳлӮҳлҠ” full specializationмқҙкі  лӢӨлҘё н•ҳлӮҳлҠ” partial specializationмһ…лӢҲлӢӨ.

    Full specialization¶

    Full specializationмқҖ м–ҙл–Ө н…ңн”ҢлҰҝмқҳ нҢҢлқјлҜён„°л“Өмқ„ м „л¶Җ лӘ…мӢңм ҒмңјлЎң м§Җм •н•ҙмЈјлҠ”, н•ҳлӮҳмқҳ мјҖмқҙмҠӨм—җ лҢҖн•ҙм„ңл§Ң м°Ёлі„нҷ”мӢңнӮӨлҠ” specializationмһ…лӢҲлӢӨ. SyntaxлҠ” лӢӨмқҢкіј к°ҷмҠөлӢҲлӢӨ.

    template <> ...

    мҳҲмӢңлҘј л“Өм–ҙліҙкІ мҠөлӢҲлӢӨ.

    #include <iostream>

template <class T> T literal(T x) { return x; }
template <> int literal(int x) { return 0; }

int main(void) {
    std::cout << literal(1) << '\n' << literal(1.0) << std::endl;
}

    literal н•ЁмҲҳлҠ” int нғҖмһ…мқҙ нҢҢлқјлҜён„°лЎң мЈјм–ҙмЎҢмқ„ л•Ңм—җ лҢҖн•ҙм„ңл§Ң лӢӨлҘҙкІҢ лҸҷмһ‘н•©лӢҲлӢӨ. (intм—җ лҢҖн•ң full specializationмқҙ м Ғмҡ©лҗЁ)

    лҳҗн•ң, explicit specializationмқ„ н•ҳкё° мң„н•ҙм„ңлҠ” мҪ”л“ң мғҒмқҳ лӢӨлҘё м–ҙл–Ө кіім—җлҸ„ лӢӨмқҢ мҪ”л“ңмІҳлҹј н•ҙлӢ№ нҢҢлқјлҜён„°мқҳ instantiationмқ„ л¶Ҳлҹ¬мҳӨлҠ” мҪ”л“ңк°Җ "лЁјм Җ" мһҲмңјл©ҙ м•Ҳ лҗ©лӢҲлӢӨ.

    #include <algorithm>
#include <vector>

template <class T> void sort(std::vector<T> &vec) {
    std::sort(vec.begin(), vec.end());
}

int main(void) {
    std::vector<int> x = {3, 2, 1};
    sort(x);
}

// template <> void sort<int>(std::vector<int> &vec) {} // Compile Error!

    лҳҗн•ң, declareл§Ң лҗҳкі  definitionмқҙ л§Ңл“Өм–ҙм§Җм§Җ м•ҠмқҖ template classлҠ” full specialization л“ұм—җ лҢҖн•ҙм„ңлҸ„ лӢӨлҘё incomplete typeмІҳлҹј мһҗмң лЎӯкІҢ мӮ¬мҡ©н•  мҲҳ мһҲмҠөлӢҲлӢӨ.

    template <class T> struct X;
template <> struct X<int>;

int main(void) {
    X<int> *ptr;
}

    Partial specialization¶

    Partial specializationмқҖ н…ңн”ҢлҰҝ нҢҢлқјлҜён„°мқҳ мқјл¶Җл§Ң specializationмқ„ н•©лӢҲлӢӨ. Function templateм—җ лҢҖн•ҙм„ңлҠ” partial specializationмқ„ н•ҳлҠ” кІғмқҙ л¶Ҳк°ҖлҠҘн•©лӢҲлӢӨ. лӢӨмқҢмқҖ мҪ”л“ң мҳҲмӢңмһ…лӢҲлӢӨ.

    #include <iostream>
#include <string>

template <class T1, class T2, int I> struct A {
    static std::string get() { return "Primary " + std::to_string(I); }
};
template <class T1, int I> struct A<T1, T1*, I> {
    static std::string get() { return "Partial " + std::to_string(I); }
};
template <class T1> struct A<T1, T1*, 3> {
    static std::string get() { return "AnotherPartial"; }
};

int main(void) {
    std::cout << A<int, double, 7>::get() << '\n';
    std::cout << A<int, int*, 7>::get() << '\n';
    std::cout << A<int, int*, 3>::get() << '\n';
}

    лҳҗн•ң, template argumentм—җлҠ” лӢӨмқҢкіј к°ҷмқҖ м ңн•ңмқҙ м Ғмҡ©лҗ©лӢҲлӢӨ.

    • Template argument listк°Җ primary templateмқҙлһ‘ мҷ„м „нһҲ лҸҷмқјн•ҙм„ңлҠ” м•Ҳ лҗ©лӢҲлӢӨ. 
      template <class A, class B> class C {};
// template <class D, class E> class C<D, E> {}; // Compile Error!
    • Default argumentк°Җ argument listм—җ лӮҳнғҖлӮ  мҲҳ м—ҶмҠөлӢҲлӢӨ.
    • Pack expansionм—җ н•ҙлӢ№н•ҳлҠ” argumentлҠ” н•ӯмғҒ argument listмқҳ л§Ҳм§Җл§үм—җ мһҲм–ҙм•ј н•©лӢҲлӢӨ.
    • м–ҙл–Ө type argumentм—җ мқҳмЎҙн•ҳлҠ” non-type argumentлҠ” specialize лҗ  мҲҳ м—ҶмҠөлӢҲлӢӨ. 
      template <class T, T t> struct A {};
// template <class T> struct A<T, 1> {}; // Compile Error!

    Template Instantiation¶

    Template н•ЁмҲҳ/нҒҙлһҳмҠӨ л“ұл“ұмқҳ м •мқҳ мһҗмІҙлҠ” м»ҙнҢҢмқј кіјм •м—җм„ң м•„л¬ҙлҹ° мҪ”л“ңлҘј мғқм„ұн•ҳм§Җ м•ҠмҠөлӢҲлӢӨ. н•ҙлӢ№ templateмқҙ мӮ¬мҡ©лҗҳлҠ” мҪ”л“ң мғҒм—җм„ң specialize лҗҳлҠ” нғҖмһ…мқ„ лӘЁл‘җ м°ҫм•„ н•ҙлӢ№ мҪ”л“ңл“Өмқ„ generateн•©лӢҲлӢӨ.

    мҪ”л“ң мғҒм—җм„ң template function, classк°Җ нҠ№м • нҢҢлқјлҜён„°(<int> л“ұ)м—җ лҢҖн•ҙ мІҳмқҢ нҳём¶ңлҗ  л•Ң, м»ҙнҢҢмқјлҹ¬лҠ” instantiationмқ„ мғқм„ұн•©лӢҲлӢӨ. Template functionмқҖ нҳём¶ңмқҙ лҗ  л•Ңл§ҲлӢӨ н•ҙлӢ№ template parameterм—җ н•ҙлӢ№н•ҳлҠ” instantiationмқ„ м°ҫм•„м„ң нҳём¶ңлҗ©лӢҲлӢӨ. InstantiationмқҖ ліҙнҶө н”„лЎңк·ёлһҳлЁё мһ…мһҘм—җм„ң көімқҙ л”°лЎң л¬ҙм–ёк°ҖлҘј н•ҳм§Җ м•Ҡм•„лҸ„ м•Ңм•„м„ң мһҳ мғқм„ұлҗҳм§Җл§Ң, мӣҗн•ңлӢӨл©ҙ explicitн•ҳкІҢ л§Ңл“Өм–ҙлӮј мҲҳ мһҲмҠөлӢҲлӢӨ.

    Explicit Instantiation¶

    Explicit instantiationмқҖ н…ңн”ҢлҰҝ лӮҙмқҳ мқёмһҗлҘј кө¬мІҙм ҒмңјлЎң м§Җм •н•ҙмӨҳм„ң auto type deducingмқ„ н•ҳм§Җ м•ҠлҠ” function template instantiationмһ…лӢҲлӢӨ. SyntaxлҠ” лӢӨмқҢкіј к°ҷмҠөлӢҲлӢӨ.

    • Template functions
    • template return_type function_name <...> (...); : Template argument deductionмқҙ м•„мҳҲ м—ҶлҠ” м •мқҳ
    • template return_type function_name (...); : Template argument deductionмқҙ нҢҢлқјлҜён„°м—җ лҢҖн•ҙм„ңл§Ң мӢӨн–үлҗҳлҠ” м •мқҳ
    • Template classes
    • template class/struct/union name <...>;

    л§Ё м•һм—җлӢӨк°Җ externмқ„ м„ нғқм ҒмңјлЎң л¶ҷмқј мҲҳлҸ„ мһҲмҠөлӢҲлӢӨ.

    Forced instantiation by expression¶

    C++ л ҲнҚјлҹ°мҠӨм—җ л”°лҘҙл©ҙ.. лӢӨмқҢкіј к°ҷмҠөлӢҲлӢӨ.

    The existence of a definition of function is considered to affect the semantics of the program if the function is needed for constant evaluation by an expression, even if constant evaluation of the expression is not required or if constant expression evaluation does not use the definition.

    кІҪмҡ°м—җ л”°лқј м–ҙл–Ө expressionмқҙ нҠ№м • н•ЁмҲҳмқҳ evaluationмқ„ н•„мҡ”лЎң н•ҳл©ҙ н•ҙлӢ№ н•ЁмҲҳмқҳ м •мқҳлҘј м»ҙнҢҢмқјлҹ¬к°Җ м°ҫлҠ”лӢӨкі  мқҙн•ҙн–ҲмҠөлӢҲлӢӨ. к·ёлһҳм„ң мҳҲмӢң мҪ”л“ңк°Җ н•ҙлӢ№ мЎ°кұҙм—җ л§ҢмЎұн•ҳлҠ” м§Җ мЎ°кёҲ л°”кҝ”м„ң н…ҢмҠӨнҠён•ҙлҙӨлҠ”лҚ°, м •л§җ м»ҙнҢҢмқј м—җлҹ¬к°Җ лңЁлҚ”кө°мҡ”.

    template<typename T>
constexpr int f() { return T::unknown_static_member; }

template<bool B, typename T>
void g(decltype(B ? f<T>() : 0)) {}

template<bool B, typename T>
void h(decltype(int{B ? f<T>() : 0})) {}

int main(void) {
    g<true, int>(0);
    // h<true, int>(0); // Compile Error, f<int> is needed
}

    decltype(B ? f<int>() : 0) м—җм„ң Bк°Җ м°ёк°’мқҙкё° л•Ңл¬ём—җ f<int>к°Җ evalлҗҳм–ҙм•ј н•ҳлҠ” кІғмІҳлҹј ліҙмқҙлҠ”лҚ°.. f<int>мқҳ return typeл§Ң ліҙкі  мӢӨм ң н•ЁмҲҳлҘј evalн•ҳлҠ” кІғмқҖ м•„лӢҲлқјлҠ” кІҢ мӢ кё°н–ҲмҠөлӢҲлӢӨ.

    Deep template instantiation depth¶

    TemplateмқҖ л§Ө instantiationл§ҲлӢӨ м„ңлЎң лӢӨлҘё мҪ”л“ңлҘј мғқм„ұн•ҳкё° л•Ңл¬ём—җ, л§ҲмқҢл§Ң лЁ№мңјл©ҙ лӢӨмқҢкіј к°ҷмқҙ м•„мЈј л§ҺмқҖ мҪ”л“ңлҘј мғқм„ұн•ҙлӮҙлҠ” кІғмқҙ к°ҖлҠҘн•©лӢҲлӢӨ.

    #include <iostream>

template <unsigned int i> void say() {
    std::cout << i << ' ';
    say<i-1>();
}

template <> void say<0>() {
    std::cout << 0 << std::endl;
}

int main(void) {
    say<1 << 20>(); // Generates 1048576, ..., 1, 0
}

    к·ёлһҳм„ң templateмқ„ мһҳлӘ» мӮ¬мҡ©н•ҳл©ҙ м»ҙнҢҢмқјлҹ¬к°Җ мғқм„ұн•  instantiationмқҳ м–‘мқҙ л„Ҳл¬ҙ м»Өм§ҖлҠ”лҚ°мҡ”. лӢӨн–үнһҲлҸ„ gccмқҳ кІҪмҡ°м—җлҠ” мқҙлҘј л§үкё° мң„н•ҙ template instantiation depth limitмқҙ мЎҙмһ¬н•©лӢҲлӢӨ. -ftemplate-depth мҳөм…ҳмқ„ мЈјл©ҙ лҗ©лӢҲлӢӨ. лӢӨлҘё м»ҙнҢҢмқјлҹ¬лҸ„ л№„мҠ·н•ң мҳөм…ҳмқҙ мһҲмқ„ кұ°лқј мғқк°Ғн•©лӢҲлӢӨ.

    мқҙкұҙ м»ҙнҢҢмқјлҹ¬л§ҲлӢӨ лӢӨлҘј мҲҳлҸ„ мһҲмқ„ кІғ к°ҷмқҖлҚ°, (м Ғм–ҙлҸ„ gcc 7.5.0 лІ„м јм—җ лҢҖн•ҙм„ңлҠ”) лӢӨмқҢкіј к°ҷмқҖ мӢқмңјлЎң specializationмқҙ лҗҳм–ҙ мһҲм§Җ м•ҠмқҖ мҪ”л“ңлҘј м»ҙнҢҢмқјлҹ¬к°Җ л¶„м„қн•ҳл©ҙ instantiationм—җ л”°лқј м•„мҳҲ нҳём¶ңлҗҳм§Җ м•ҠлҠ” мҪ”л“ңк°Җ мһҲлҚ”лқјлҸ„ semanticм—җ мҳҒн–Ҙмқҙ к°„лӢӨкі  л¶„м„қн•ҳлҠ” л“Ҝ н•©лӢҲлӢӨ. лӢӨмқҢ мҪ”л“ңлҠ” м»ҙнҢҢмқјмӢң template instantiation depth limit exceeded м—җлҹ¬к°Җ л°ңмғқн•ҳл©°, underflowк°Җ мқјм–ҙлӮҳлҠ” кІғмқ„ ліҙмӢӨ мҲҳ мһҲмҠөлӢҲлӢӨ.

    template <unsigned int I> void recursive() {
    if(I == 0) return;
    else recursive<I-1>();
}

int main(void) {
    recursive<3>();
}

    Variadic Templates (Parameter Pack)¶

    Variadic templates лҳҗлҠ” Parameter packмқҙлһҖ, мһ„мқҳмқҳ к°ңмҲҳмқҳ н…ңн”ҢлҰҝ нҢҢлқјлҜён„°лҘј к°Җм§ҖлҠ” н…ңн”ҢлҰҝмқ„ мқҳлҜён•©лӢҲлӢӨ. SyntaxлҠ” лӢӨмқҢкіј к°ҷмҠөлӢҲлӢӨ.

    template <something T...> blabla;

    лӢӨмқҢ мҪ”л“ңлҠ” н…ңн”ҢлҰҝ нҢҢлқјлҜён„°л“Өмқҳ н•©мқ„ кө¬н•ҳлҠ” sum functionмқҳ мҳҲмӢңмһ…лӢҲлӢӨ. sum1мқҖ к°ҷмқҖ нғҖмһ… T нҢҢлқјлҜён„°лҘј лӘҮ к°ңл“  л°ӣмқ„ мҲҳ мһҲмҠөлӢҲлӢӨ.

    #include <iostream>

template <class T> T sum1() { return T{0}; }
// template <class T, T v1> T sum1() { return v1; } // Ambiguous!
template <class T, T v1, T... v2> T sum1() {
    return v1 + sum1<T, v2...>();
}

int main(void) {
    std::cout << sum1<int, 1>() << '\n';
    std::cout << sum1<int, 1, 2>() << '\n';
    std::cout << sum1<int, 1, 2, 3>() << '\n';
}

    лӢӨмқҢ мҪ”л“ңлҠ” нҢҢлқјлҜён„°л“Өмқ„ printн•ҳлҠ” print functionмқҳ мҳҲмӢңмһ…лӢҲлӢӨ. printлҠ” м„ңлЎң лӢӨлҘё нғҖмһ…(Types...)мқҳ нҢҢлқјлҜён„°лҘј лӘҮ к°ңл“  л°ӣмқ„ мҲҳ мһҲмҠөлӢҲлӢӨ.

    #include <iostream>

template <class T> void print(std::ostream &out, T value) {
    out << value << std::endl;
}

template <class T, class... Types> void print(std::ostream &out, T value, Types... values) {
    out << value << ' ';
    print(out, values...);
}

int main(void) {
    print(std::cout, 1, 2.3, "abc");
}

    Fold Expression¶

    Fold expressionмқҙлһҖ, binary operator н•ҳлӮҳлЎң parameter packмқҳ лӘЁл“  мӣҗмҶҢлҘј м—°мӮ°мӢңнӮӨлҠ” кІғмқ„ мқҳлҜён•©лӢҲлӢӨ. Syntax л°Ҹ к°Ғ syntaxм—җ м—°кІ°лҗҳлҠ” кІ°кіјлҠ” лӢӨмқҢкіј к°ҷмҠөлӢҲлӢӨ.

    1. (E op ...) -> E1 op (E2 op ( ... (E[N-1] op E[N])))
    2. (... op E) -> ((E1 op E2) op ... ) op E[N]
    3. (E op ... op I) -> E1 op (E2 op ... (E[N-1] op (E[N] op I)))
    4. (I op ... op E) -> (((I op E1) op E2) op ... ) op E[N]

    ліҙнҶө лҸҷмқјн•ң м—°мӮ°мһҗлҘј м—¬лҹ¬ лІҲ м Ғмҡ©н•ҳлҠ” м—°мӮ°мқҳ кІҪмҡ° мҷјмӘҪл¶Җн„° мҳӨлҘёмӘҪмңјлЎң accumulateн•ҳлҠ” кІҪмҡ°к°Җ л§ҺмқҖлҚ°, к·ёкІҢ 2лІҲ, 4лІҲмқҳ syntaxлқјлҠ” м җмқ„ мЈјмқҳн•ҳм…”м•ј н•  кІғ к°ҷмҠөлӢҲлӢӨ.

    лӢӨмқҢ мҪ”л“ңлҠ” fold expressionмқ„ мӮ¬мҡ©н•ҳлҠ” мҳҲмӢңмһ…лӢҲлӢӨ.

    #include <iostream>

template <class... Args> int any(Args... bools) {
    return (int)(... || bools);
}

int main(void) {
    std::cout << any(false, false, false) << any(false, false, true) << '\n';
}

    Fold expressionмқ„ мӮ¬мҡ©н•ҳмӢӨ л•ҢлҠ” лӘҮ к°Җм§ҖлҘј мЈјмқҳн•ҳм…”м•ј н•©лӢҲлӢӨ.

    1. Fold expression л°”к№Ҙм—җ л°ҳл“ңмӢң syntaxм—җ мҚЁм§„ лҢҖлЎң кҙ„нҳёлҘј мҚЁмӨҳм•ј н•©лӢҲлӢӨ.
    2. Fold expressionмқҳ init к°’м—җ лӢЁмҲң к°’мқҙ м•„лӢҢ мҲҳмӢқмқ„ л„Јмқ„ кІҪмҡ°, кҙ„нҳёлЎң к°җмӢёмЈјм–ҙм•ј н•©лӢҲлӢӨ. 
      #include <iostream>

template <int Init, class... Args>
int sum_with_double_init(Args... args) {
    // return (Init * 2 + ... + args);
    return ((Init * 2) + ... + args);
}

int main(void) {
    std::cout << sum_with_double_init<5>(1, 2, 3) << '\n';
}

    мқҙмғҒмңјлЎң templateм—җ кҙҖн•ҳм—¬ м—¬лҹ¬ к°Җм§ҖлҘј л§Һмқҙ мӮҙнҺҙліҙм•ҳмҠөлӢҲлӢӨ! к·јлҚ° м•„м§Ғ л№ м§„ лӮҙмҡ©мқҙ мһҲмқ„ мҲҳ мһҲм–ҙ м¶”нӣ„м—җ нҸ¬мҠӨнҠёлҘј ліҙмҷ„н•  мҲҳ мһҲмҠөлӢҲлӢӨ. мқҙмғҒмңјлЎң, C++ мӢңлҰ¬мҰҲ 3лІҲм§ё нҸ¬мҠӨнҢ…мқ„ л§Ҳм№ҳкІ мҠөлӢҲлӢӨ. к°җмӮ¬н•©лӢҲлӢӨ!
